Daten aufbereiten:

  • Datumsformat
  • gebürtige Obwaldener identifizieren
  • letztes Steuerjahr pro Person erzeugen (Variable ENDE)
  • Variable für Personen deren Steuerpflicht ohne Wegzug in eine andere schweizer Gemeinde endet (verstorben oder Wegzug ins Ausland)
  • Wohnsitz Variable: Zeigt an, ob die Person in einer der 7 obwaldener Gemeinden wohnt (man könnte noch Restschweiz und Ausland unterscheiden in Zukunft)
daten$ZUZUG <- as.Date(daten$ZUZUGDATUM)
daten$ZUZUGJAHR <- as.numeric(substr(daten$ZUZUGDATUM,1,4))
daten$HIESIG <- factor(daten$ZUZUGJAHR==daten$JAHRGANG,label=c("neu","gebürtiger Obwaldener"))
daten <- daten %.% group_by(PERSID) %.% mutate(ENDE=max(STEUERJAHR))
daten$TOT_ODER_AUSLAND <- factor(is.na(daten$WEGZUGNACHBFS)&is.na(daten$WEGZUGKT)&daten$ENDE<2013,label=c("Wegzug innerhalb CH","tot oder Ausland"))
daten$WEGZUG <- ifelse(!is.na(daten$WEGZUGNACHBFS)|!is.na(daten$WEGZUGKT), daten$ENDE, NA)
# Personen mit Wohnsitz in OW
daten$WOHNSITZ_OW <- factor(!is.na(substring(str_extract(daten[,"WOHNORT"], "Alpnach|Engelberg|Giswil|Kerns|Lungern|Sachseln|Sarnen"),1)),label=c("auswärtig","Wohnsitz OW"))
